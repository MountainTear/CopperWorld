在随机坐标生成瓦片

```C#
using System.Collections;
using UnityEngine;
using UnityEngine.Tilemaps;

public class TileMapController : MonoBehaviour
{
    public Tilemap tilemap;
    public TileBase[] tiles; // 要添加的瓦片

    private void Start()
    {
        StartCoroutine(AddTileRoutine());
    }

    private IEnumerator AddTileRoutine()
    {
        while (true)
        {
            yield return new WaitForSeconds(1f); // 每隔1秒执行一次

            Vector3Int randomPosition = GetRandomTilePosition(); // 获取随机坐标点
            tilemap.SetTile(randomPosition, GetRandomTile()); // 在随机坐标点添加随机瓦片
        }
    }

    private Vector3Int GetRandomTilePosition()
    {
        // 在TileMap范围内获取随机坐标点
        BoundsInt bounds = tilemap.cellBounds;
        return new Vector3Int(
            Random.Range(bounds.min.x, bounds.max.x),
            Random.Range(bounds.min.y, bounds.max.y),
            0
        );
    }

    private TileBase GetRandomTile()
    {
        // 从tiles数组中随机选择一个瓦片
        return tiles[Random.Range(0, tiles.Length)];
    }
}
```
